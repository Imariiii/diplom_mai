\newglossaryentry{term_verify_restore}{
    name={Верификация восстановления},
    description={проверка целостности базы данных после тестового прогона путём сравнения числа строк и, при необходимости, контрольных сумм таблиц с зафиксированными до теста значениями; подтверждает корректность автоматического восстановления состояния}
}
\newglossaryentry{term19}{
    name={Взаимная блокировка},
    description={ситуация при конкурентном доступе к данным, при которой два или более параллельных сеанса взаимно удерживают ресурсы, необходимые друг другу, что приводит к принудительному завершению одного из них СУБД и снижению общей пропускной способности}
}
\newglossaryentry{term6}{
    name={Виртуальный пользователь},
    description={эмулированная сущность, выполняющая операции нагрузки в ходе нагрузочного теста~--- одиночные SQL-запросы или атомарные транзакционные последовательности~--- для имитации реальной рабочей нагрузки; число виртуальных пользователей задаёт уровень конкурентности}
}
\newglossaryentry{term8}{
    name={Временной ряд},
    description={последовательность значений метрик производительности, зафиксированных через равные интервалы времени в ходе выполнения теста и используемых для построения графиков в реальном времени}
}
\newglossaryentry{term12}{
    name={Группа баз данных},
    description={именованная группа физических подключений к СУБД, объединённых единым профилем схемы; используется как единица конфигурации при одновременном нагрузочном тестировании нескольких экземпляров СУБД с одинаковой структурой данных}
}
\newglossaryentry{term4}{
    name={Задержка},
    description={время от отправки запроса к базе данных до получения ответа, измеряемое в миллисекундах}
}
\newglossaryentry{term18}{
    name={Коэффициент попаданий в кэш},
    description={доля запросов к базе данных, обработанных без обращения к дисковому хранилищу за счёт наличия данных в буферном пуле; служит индикатором эффективности использования оперативной памяти СУБД}
}
\newglossaryentry{term11}{
    name={Набор сценария нагрузки},
    description={конкретная конфигурация нагрузки для пары <<профиль схемы + шаблон сценария>>: перечень параметризованных SQL-запросов или транзакционных последовательностей операций изменения данных с весами выполнения и спецификацией индексов; фиксируется в неизменяемом снимке при запуске прогона}
}
\newglossaryentry{term1}{
    name={Нагрузочное тестирование},
    description={вид тестирования производительности, направленный на определение поведения системы под ожидаемой и повышенной нагрузкой}
}
\newglossaryentry{term5}{
    name={Процентиль},
    description={значение, ниже которого находится определённый процент измерений в наборе данных (например, p95 означает, что 95\% запросов выполнены быстрее данного значения)}
}
\newglossaryentry{term13}{
    name={Профиль схемы},
    description={описание структуры данных, общей для группы совместимых баз данных одной предметной области; служит основой для автоматической генерации наборов сценариев нагрузки}
}
\newglossaryentry{term20}{
    name={Прогрев},
    description={подготовительная фаза нагрузочного тестирования, состоящая из подфазы наращивания нагрузки и подфазы удержания на заданном уровне; метрики этой фазы не включаются в итоговый статистический анализ}
}
\newglossaryentry{term3}{
    name={Пропускная способность},
    description={количество успешно обработанных единиц нагрузки в единицу времени: запросов для сценариев с одиночными SQL-операциями или транзакций для транзакционных сценариев}
}
\newglossaryentry{term14}{
    name={Размер эффекта},
    description={стандартизированная мера практической значимости различий между двумя выборками; в работе используется коэффициент Коэна~$d$, интерпретируемый как пренебрежимый при $d < 0{,}2$, малый при $0{,}2 \leq d < 0{,}5$, средний при $0{,}5 \leq d < 0{,}8$ и большой при $d \geq 0{,}8$}
}
\newglossaryentry{term2}{
    name={Система управления базами данных},
    description={программное обеспечение для создания, ведения и управления базами данных, обеспечения доступа к ним и контроля целостности}
}
\newglossaryentry{term9}{
    name={Снимок состояния базы данных},
    description={фиксация состояния тестируемой базы данных перед прогоном с операциями записи: число строк в таблицах и, при включённой настройке, контрольные суммы данных; используется для последующей верификации восстановления}
}
\newglossaryentry{term_compare}{
    name={Сравнительный анализ},
    description={автоматизированное сопоставление результатов двух и более тестовых прогонов с вычислением описательной статистики, проверкой статистической значимости различий и формированием аналитического отчёта; поддерживает попарное сравнение СУБД при фиксированной нагрузке и серийный анализ при изменении числа виртуальных пользователей}
}
\newglossaryentry{term_significance}{
    name={Статистическая значимость},
    description={вероятностная оценка того, что наблюдаемое различие метрик между сравниваемыми прогонами не обусловлено случайной вариацией; в системе определяется по p-value выбранного критерия с учётом поправки на множественные сравнения}
}
\newglossaryentry{term15}{
    name={Точка насыщения},
    description={уровень нагрузки, при превышении которого дальнейшее увеличение нагрузки не даёт прироста пропускной способности, а лишь увеличивает задержку}
}
\newglossaryentry{term22}{
    name={Транзакционный набор сценария нагрузки},
    description={набор сценария нагрузки для OLTP-шаблона, заданный вручную в виде последовательностей связанных операций изменения данных, выполняемых атомарно в рамках одной транзакции; в отличие от автогенерируемых наборов не формируется по схеме базы данных автоматически}
}
\newglossaryentry{term23}{
    name={Фаза измерения},
    description={основная фаза нагрузочного теста, следующая после завершения прогрева; метрики, собранные в измерительной фазе, используются для статистического анализа и сравнения прогонов}
}
\newglossaryentry{term16}{
    name={Хвостовая задержка},
    description={значение задержки на высоких перцентилях распределения (p95, p99), характеризующее поведение системы при обработке наиболее медленных запросов; является ключевым показателем стабильности производительности при конкурентной нагрузке}
}
\newglossaryentry{term_template}{
    name={Шаблон сценария},
    description={логическое описание вида нагрузки без привязки к конкретной схеме данных; в системе предусмотрено шесть встроенных шаблонов (только чтение, только запись, смешанная лёгкая или тяжёлая нагрузка, OLTP, аналитические запросы); для пяти из них набор сценария нагрузки формируется автоматически по схеме, для OLTP~--- из встроенных транзакционных конфигураций}
}
